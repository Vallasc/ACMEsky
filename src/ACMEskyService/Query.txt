SELECT f FROM Flight f, FlightInterest fi, UserInterest ui
WHERE 
ui.id = "prova user interest" AND
ui.outboundFlightInterest = fi.id AND 
fi.departure_airport_id = f.departure_airport_id AND
fi.arrival_airport_id = f.arrival_airport_id AND
fi.departure_date_time = f.departure_date_time AND
f.booked = false
